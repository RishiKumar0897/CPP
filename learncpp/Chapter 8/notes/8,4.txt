- constexpr if statement: requires conditional to be a constexpr 
    - conditional is evaluated at runtime 

#include <iostream>

int main()
{
	constexpr double gravity{ 9.8 };

	if constexpr (gravity == 9.8) // now using constexpr if
		std::cout << "Gravity is normal.\n";
	else
		std::cout << "We are not on Earth.\n";

	return 0;
}



- favor constexpr if statements over non constexpr if statements when the conditional is a constant expression
